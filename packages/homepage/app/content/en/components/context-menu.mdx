## Contract

`TRContextMenu` attaches commands to content such as a rack row, file, or canvas. Use `TRMenu` when a dedicated trigger always opens the command list. This Tinyrack wrapper preserves Base UI positioning, focus, dismissal, React 19 refs, native props, events, and the `render` contract. Playground controls cover `label`, `open`, and `disabledItem`.

## Usage

<ComponentExampleTabs
  id="context-menu-basic"
  title="Rack row actions"
  description="Right-click the rack row or focus it and press Shift+F10. The separate More actions button opens the same commands for touch users."
  preview={<Stories.ContextMenuPreview label="Rack Alpha" open={false} disabledItem={false} />}
  sources={[{ label: 'React', language: 'tsx', code: String.raw`import { TRButton } from '@tinyrack/ui/components/button';
import { TRContextMenu } from '@tinyrack/ui/components/context-menu';
import { ChevronRight, Server } from 'lucide-react';
import { useRef } from 'react';

function RackRowActions() {
  const triggerRef = useRef<HTMLDivElement>(null);

  function openActions() {
    const rect = triggerRef.current?.getBoundingClientRect();
    if (!triggerRef.current || !rect) return;
    triggerRef.current.dispatchEvent(new MouseEvent('contextmenu', {
      bubbles: true,
      button: 2,
      clientX: rect.right - 24,
      clientY: rect.top + rect.height / 2,
    }));
  }

  return (
    <div className="grid gap-2 sm:grid-cols-[minmax(0,1fr)_auto]">
      <TRContextMenu.Root>
        <TRContextMenu.Trigger
          aria-label="Rack Alpha, online rack. Open context menu for actions."
          className="grid w-full grid-cols-[auto_minmax(0,1fr)_auto] items-center gap-3"
          ref={triggerRef}
          render={<TRButton appearance="ghost" type="button" />}
        >
          <Server aria-hidden="true" />
          <span>Rack Alpha</span>
          <span>10.42.0.18 · Seoul</span>
          <span>Online</span>
        </TRContextMenu.Trigger>
        <TRButton appearance="outline" onClick={openActions} uiSize="sm">
          More actions
        </TRButton>
        <TRContextMenu.Portal>
          <TRContextMenu.Backdrop />
          <TRContextMenu.Positioner>
            <TRContextMenu.Popup>
              <TRContextMenu.Arrow />
              <TRContextMenu.Group>
                <TRContextMenu.GroupLabel>Rack Alpha</TRContextMenu.GroupLabel>
                <TRContextMenu.LinkItem href="/racks/alpha">Open details</TRContextMenu.LinkItem>
                <TRContextMenu.Item>Copy address</TRContextMenu.Item>
                <TRContextMenu.Item>Restart</TRContextMenu.Item>
              </TRContextMenu.Group>
              <TRContextMenu.Separator />
              <TRContextMenu.SubmenuRoot>
                <TRContextMenu.SubmenuTrigger>
                  Move to <ChevronRight aria-hidden="true" />
                </TRContextMenu.SubmenuTrigger>
                <TRContextMenu.Portal>
                  <TRContextMenu.Positioner>
                    <TRContextMenu.Popup>
                      <TRContextMenu.Arrow />
                      <TRContextMenu.Item>Production</TRContextMenu.Item>
                      <TRContextMenu.Item>Staging</TRContextMenu.Item>
                    </TRContextMenu.Popup>
                  </TRContextMenu.Positioner>
                </TRContextMenu.Portal>
              </TRContextMenu.SubmenuRoot>
              <TRContextMenu.Separator />
              <TRContextMenu.Item variant="danger">Remove rack</TRContextMenu.Item>
            </TRContextMenu.Popup>
          </TRContextMenu.Positioner>
        </TRContextMenu.Portal>
      </TRContextMenu.Root>
    </div>
  );
}` }]}
/>

## API

Every compound part and prop/state type is exported from `@tinyrack/ui/components/context-menu`; styles remain opt-in through the matching CSS subpath. Use `Item variant="danger"` for destructive commands. `Backdrop`, `Positioner`, `Popup`, and `Arrow` share the layer anatomy; `Popup` is also the constrained scrolling viewport. Make the target focusable for Shift+F10 and provide an explicit control that dispatches the same context-menu action when touch users cannot right-click.
