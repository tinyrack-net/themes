## Contract

| Axis | Values |
| --- | --- |
| state | unchecked, checked, `indeterminate` |
| availability | enabled, `readOnly`, `disabled` |
| form | `name`, `value`, `required`, hidden input validity |

Use `uiSize="sm"` for a smaller checkbox in dense control surfaces.

  const [attempted, setAttempted] = useState(false);
  const [checked, setChecked] = useState(false);
  const invalid = attempted && !checked;

  return (
    <TRForm noValidate onSubmit={(event) => { event.preventDefault(); setAttempted(true); }}>
      <TRField.Root invalid={invalid}>
        <TRField.Label>
          <TRCheckbox.Root checked={checked} name="terms" onCheckedChange={setChecked} required>
            <TRCheckbox.Indicator aria-hidden="true">✓</TRCheckbox.Indicator>
          </TRCheckbox.Root>
          I accept the maintenance window.
        </TRField.Label>
        {invalid ? <TRField.Error match>Accept the maintenance window to continue.</TRField.Error> : null}
      </TRField.Root>
      <TRButton type="submit">Continue</TRButton>
    </TRForm>
  );
}` }]}
/>

<ComponentExampleTabs
  id="checkbox-form-values"
  title="External form and explicit off value"
  description="The checkbox is rendered outside its owner form. `value` and `uncheckedValue` make both on and off states explicit in FormData."
  preview={<Stories.CheckboxFormValuesPreview />}
  sources={[{ label: 'React', language: 'tsx', code: String.raw`import { useState } from 'react';
import { TRButton } from '@tinyrack/ui/components/button';
import { TRCheckbox } from '@tinyrack/ui/components/checkbox';
import { TRForm } from '@tinyrack/ui/components/form';

export function MonitoringSetting() {
  const [result, setResult] = useState('');

  return <>
    <TRForm id="monitoring-form" onSubmit={(event) => {
      event.preventDefault();
      setResult(String(new FormData(event.currentTarget).get('monitoring')));
    }}>
      <TRButton type="submit">Read form value</TRButton>
    </TRForm>
    <label>
      <TRCheckbox.Root
        form="monitoring-form"
        name="monitoring"
        uncheckedValue="disabled"
        value="enabled"
      >
        <TRCheckbox.Indicator aria-hidden="true">✓</TRCheckbox.Indicator>
      </TRCheckbox.Root>
      Monitoring
    </label>
    <output aria-live="polite">{result}</output>
  </>;
}` }]}
/>

## API

`Root`, `Indicator`, their prop types, and Base UI state types are exported from `@tinyrack/ui/components/checkbox`; CSS remains opt-in.
