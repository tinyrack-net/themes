# Switch

Toggles an immediate binary setting while preserving native checked state and form semantics.

## Contract

| Axis | Behavior |
| --- | --- |
| state | uncontrolled with `defaultChecked`, controlled with `checked` and `onCheckedChange` |
| semantics | pair the switch with a visible label; use a checkbox when submitting a batch choice |
| form | supports `name`, `value`, and `required` through its hidden native input |

  const [attempted, setAttempted] = useState(false);
  const [checked, setChecked] = useState(false);
  const invalid = attempted && !checked;

  return (
    <Form noValidate onSubmit={(event) => {
      event.preventDefault();
      event.currentTarget.checkValidity();
      setAttempted(true);
    }}>
      <Field.Root invalid={invalid}>
        <Field.Label>
          <Switch.Root checked={checked} name="acknowledged" onCheckedChange={setChecked} required>
            <Switch.Thumb />
          </Switch.Root>
          Acknowledge the maintenance window
        </Field.Label>
        <Field.Error match>Acknowledge the window to continue.</Field.Error>
      </Field.Root>
      <Button type="submit">Continue</Button>
    </Form>
  );
}` }]}
/>
