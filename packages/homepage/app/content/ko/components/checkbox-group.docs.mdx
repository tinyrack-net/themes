## 계약

| 축 | 동작 |
| --- | --- |
| value | uncontrolled with `defaultValue`, controlled with `value` and `onValueChange` |
| form | give each checkbox the same `name`; every checked item contributes its `value` |
| availability | group-level `disabled` cascades; individual checkboxes can be `readOnly` |
| accessibility | name the group and every `TRCheckbox.Root`; a native-input `htmlFor` alone does not name the visual checkbox role |
| validation | compose controlled values with TRForm/TRField rules when the product requires a minimum or maximum selection count |

  sources={[{ label: 'React', language: 'tsx', code: String.raw`const featureOptions = [
  ['metrics', 'Metrics'],
  ['alerts', 'Alerts'],
  ['backups', 'Automated backups'],
];

function FeatureForm() {
  const [attempted, setAttempted] = useState(false);
  const [selected, setSelected] = useState(['metrics']);
  const [submitted, setSubmitted] = useState('');
  const invalid = attempted && (selected.length < 1 || selected.length > 2);

  return (
    <TRForm onSubmit={(event) => {
      event.preventDefault();
      setAttempted(true);
      const values = new FormData(event.currentTarget).getAll('features');
      setSubmitted(values.length >= 1 && values.length <= 2 ? values.join(', ') : '');
    }}>
      <TRCheckboxGroup
        aria-describedby={invalid ? 'feature-error' : undefined}
        aria-invalid={invalid || undefined}
        aria-label="Included features"
        onValueChange={(values) => { setSelected(values); setSubmitted(''); }}
        value={selected}
      >
        {featureOptions.map(([value, label]) => {
          const inputId = 'feature-' + value;
          const labelId = inputId + '-label';
          return <div key={value}>
            <TRCheckbox.Root
              aria-labelledby={labelId}
              id={inputId}
              name="features"
              value={value}
            >
              <TRCheckbox.Indicator aria-hidden="true">✓</TRCheckbox.Indicator>
            </TRCheckbox.Root>
            <label htmlFor={inputId} id={labelId}>{label}</label>
          </div>;
        })}
      </TRCheckboxGroup>
      {invalid ? <p id="feature-error" role="alert">
        {selected.length < 1 ? 'TRSelect at least one feature.' : 'TRSelect no more than two features.'}
      </p> : null}
      <TRButton type="submit">Save features</TRButton>
      <output aria-live="polite">{submitted}</output>
    </TRForm>
  );
}` }]}
/>

## API

`TRCheckboxGroup`과 prop/state type은 `@tinyrack/ui/components/checkbox-group`에서 export됩니다. 각 시각적 checkbox role에 고유한 `value` prop과 접근 가능한 이름을 가진 `TRCheckbox.Root` children을 렌더링하세요.
