## コントラクト

| Axis | Values |
| --- | --- |
| state | unchecked, checked, `indeterminate` |
| availability | enabled, `readOnly`, `disabled` |
| form | `name`, `value`, `required`, hidden input validity |

密度の高いコントロールサーフェスで小さな checkbox が必要な場合は `uiSize="sm"` を使います。

  const [attempted, setAttempted] = useState(false);
  const [checked, setChecked] = useState(false);
  const invalid = attempted && !checked;

  return (
    <Form noValidate onSubmit={(event) => { event.preventDefault(); setAttempted(true); }}>
      <Field.Root invalid={invalid}>
        <Field.Label>
          <Checkbox.Root checked={checked} name="terms" onCheckedChange={setChecked} required>
            <Checkbox.Indicator aria-hidden="true">✓</Checkbox.Indicator>
          </Checkbox.Root>
          I accept the maintenance window.
        </Field.Label>
        {invalid ? <Field.Error match>Accept the maintenance window to continue.</Field.Error> : null}
      </Field.Root>
      <Button type="submit">Continue</Button>
    </Form>
  );
}` }]}

import { Button } from '@tinyrack/ui/components/button';
import { Checkbox } from '@tinyrack/ui/components/checkbox';
import { Form } from '@tinyrack/ui/components/form';

export function MonitoringSetting() {
  const [result, setResult] = useState('');

  return <>
    <Form id="monitoring-form" onSubmit={(event) => {
      event.preventDefault();
      setResult(String(new FormData(event.currentTarget).get('monitoring')));
    }}>
      <Button type="submit">Read form value</Button>
    </Form>
    <label>
      <Checkbox.Root
        form="monitoring-form"
        name="monitoring"
        uncheckedValue="disabled"
        value="enabled"
      >
        <Checkbox.Indicator aria-hidden="true">✓</Checkbox.Indicator>
      </Checkbox.Root>
      Monitoring
    </label>
    <output aria-live="polite">{result}</output>
  </>;
}` }]}
/>

## API

`Root`、`Indicator`、それらの prop type、Base UI state type は `@tinyrack/ui/components/checkbox` から export されます。CSS は opt-in のままです。
