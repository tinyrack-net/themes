## コントラクト

| Concern | Contract |
| --- | --- |
| submission | native `submit` events and `FormData`, or parsed field values through `onFormSubmit` |
| validation | native validity plus external `errors` rendered by Base UI `Field.Error` composition |
| reset | native `type="reset"` restores uncontrolled values; clear application results and server errors in `onReset` |
| state | applications own input values and post-submit results |

  const [value, setValue] = useState('');
  const invalid = attempted && value.trim().length === 0;

  return (
    <Form
      noValidate
      onSubmit={(event) => {
        event.preventDefault();
        setAttempted(true);
        if (event.currentTarget.checkValidity()) setSaved(value);
      }}
    >
      <label htmlFor="required-rack">Rack name</label>
      <Input
        aria-describedby={invalid ? 'required-rack-error' : undefined}
        aria-invalid={invalid || undefined}
        id="required-rack"
        name="rack"
        onChange={(event) => setValue(event.currentTarget.value)}
        required
        value={value}
      />
      {invalid ? <p id="required-rack-error" role="alert">Enter a rack name before saving.</p> : null}
      <Button type="submit">Save rack</Button>
      <output aria-live="polite">{saved ? 'Saved ' + saved + '.' : ''}</output>
    </Form>
  );
}` }]}

  const [result, setResult] = useState('');

  return <Form
    aria-label="Create rack form"
    errors={errors}
    onFormSubmit={(values) => {
      const rack = String(values.rack ?? '');
      if (rack.toLowerCase() === 'rack-alpha') {
        setErrors({ rack: 'Rack Alpha already exists.' });
        setResult('');
        return;
      }
      setErrors({});
      setResult('Created ' + rack + '.');
    }}
    onReset={() => { setErrors({}); setResult(''); }}
  >
    <Field.Root name="rack">
      <Field.Label>Rack name</Field.Label>
      <Field.Control defaultValue="rack-alpha" required />
      <Field.Description>Use a name that is not already registered.</Field.Description>
      <Field.Error />
    </Field.Root>
    <Button type="submit">Create rack</Button>
    <Button type="reset">Reset form</Button>
    <output aria-live="polite">{result}</output>
  </Form>;
}` }]}

## API

`Form` はネイティブ form props、ref、event、`className`、`style`、`render` を転送し、control は自身の value を管理します。`FormProps<Values>` は `onFormSubmit` を通じて field-value generic を保持します。server response には field name と `errors` を使い、Form または Field スコープで `validationMode` を選びます。submit を生成せず明示的に検証するには `FormActions.validate(name?)` を使います。
