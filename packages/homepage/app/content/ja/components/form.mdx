## コントラクト

| 項目 | コントラクト |
| --- | --- |
| 送信 | ネイティブの `submit` イベントと `FormData`、または `onFormSubmit` で解析されたフィールド値を使用します。 |
| 検証 | ネイティブ validity と、Base UI の `TRField.Error` コンポジションで表示する外部 `errors` を組み合わせます。 |
| リセット | ネイティブの `type="reset"` は非制御値を復元します。制御値とアプリケーション結果は `onReset` でリセットします。 |
| 状態 | control は非制御またはアプリケーション制御で利用でき、server error と送信結果はアプリケーションが所有します。 |

function NativeRackForm() {
  const [submitted, setSubmitted] = useState('');

    aria-label="ネイティブのラックフォーム"
    onReset={() => setSubmitted('')}

  </TRForm>;

  description="空のまま submit してエラーを確認し、rack name を入力してもう一度 submit し成功を確認します。"

import '@tinyrack/ui/core.css';
import '@tinyrack/ui/components/button.css';
import '@tinyrack/ui/components/form.css';
import '@tinyrack/ui/components/input.css';
import { TRButton } from '@tinyrack/ui/components/button';
import { TRForm } from '@tinyrack/ui/components/form';
import { TRInput } from '@tinyrack/ui/components/input';

function RequiredRackForm() {
  const [attempted, setAttempted] = useState(false);
  const [saved, setSaved] = useState('');
  const [value, setValue] = useState('');
  const invalid = attempted && value.trim().length === 0;

  return (
    <TRForm
      noValidate
      onSubmit={(event) => {
        event.preventDefault();
        setAttempted(true);
        if (event.currentTarget.checkValidity()) setSaved(value);
      }}
    >
      <label htmlFor="required-rack">ラック名</label>
      <TRInput
        aria-describedby={invalid ? 'required-rack-error' : undefined}
        aria-invalid={invalid || undefined}
        id="required-rack"
        name="rack"
        onChange={(event) => {
          setValue(event.currentTarget.value);
          setSaved('');
        }}
        required
        value={value}
      />
      {invalid ? <p id="required-rack-error" role="alert">保存する前にラック名を入力してください。</p> : null}
      <TRButton type="submit">ラックを保存</TRButton>
      <output aria-live="polite">{saved ? saved + ' を保存しました。' : ''}</output>
    </TRForm>
  );
}` }]}
/>

  const [errors, setErrors] = useState<Record<string, string>>({});
  const [result, setResult] = useState('');

  return <TRForm

      <TRField.Label>ラック名</TRField.Label>
      <TRField.Control defaultValue="rack-alpha" required />
      <TRField.Description>未登録の名前を使ってください。</TRField.Description>
      <TRField.Error />
    </TRField.Root>
    <TRButton type="submit">ラックを作成</TRButton>
    <TRButton type="reset">フォームをリセット</TRButton>
    <output aria-live="polite">{result}</output>
  </TRForm>;
}` }]}

export function RackForm() {
  const actionsRef = useRef<TRFormActions>(null);
  const [result, setResult] = useState('');

  return <TRForm<{ rack: string; region: string }>
    actionsRef={actionsRef}
    onFormSubmit={(values) => setResult(values.region + ' の ' + values.rack + ' を検証しました。')}

## API

`TRForm` はネイティブ form props、ref、event、`className`、`style`、`render` を転送し、control は自身の value を管理します。`TRFormProps<Values>` は `onFormSubmit(values, details)` を通じて field-value generic を保持し、event detail には `TRFormSubmitEventDetails` と `TRFormSubmitEventReason` を利用できます。server response には field name と `errors` を使い、TRForm または TRField スコープで `validationMode` を選びます。submit を生成せず明示的に検証するには `TRFormActions.validate(name?)` を使います。native reset が復元するのは非制御 control のみなので、制御値はアプリケーションの `onReset` handler でリセットしてください。
