## コントラクト

| 軸 | 値 | 既定値 |
| --- | --- | --- |
| selection | controlled `value` or uncontrolled `defaultValue` | `[]` |
| expansion | single or `multiple` | single |
| availability | root `disabled` or item `disabled` | interactive |
| lifecycle | unmount, `keepMounted`, or `hiddenUntilFound` | unmount |

import { Accordion } from '@tinyrack/ui/components/accordion';

export function AccordionExample() {
  const [value, setValue] = useState<string[]>(['overview']);

  return (
    <div className="grid gap-3">
      <Accordion.Root multiple onValueChange={setValue} value={value}>
        <Accordion.Item value="overview">
          <Accordion.Header>
            <Accordion.Trigger>What is Tinyrack?</Accordion.Trigger>
          </Accordion.Header>
          <Accordion.Panel>A React-only UI system.</Accordion.Panel>
        </Accordion.Item>
        <Accordion.Item value="install">
          <Accordion.Header>
            <Accordion.Trigger>How do I install it?</Accordion.Trigger>
          </Accordion.Header>
          <Accordion.Panel>Install the package and component CSS.</Accordion.Panel>
        </Accordion.Item>
      </Accordion.Root>
      <output aria-live="polite">
        Expanded: {value.length === 0 ? 'none' : value.join(', ')}
      </output>
    </div>
  );
}` }]}

## API

Base UI が展開、制御付き・非制御状態、trigger の意味、panel の関係を担当します。Accordion の trigger は通常の document tab 順序に残し、非推奨の矢印キー roving-focus props はこのガイドに含めません。
