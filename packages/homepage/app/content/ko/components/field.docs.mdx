## 계약

| Part | Responsibility |
| --- | --- |
| `Root` | disabled, invalid, dirty, touched, and validation state |
| `Label` | accessible name and enlarged target |
| `Control` | native input rendered through Base UI |
| `Description` | persistent instructions |
| `Error` | validity-specific recovery message |
| `Validity` | render-prop access to native validity |
| `Item` | repeatable option within a collection field |

<ComponentExampleTabs
  id="field-validation"
  title="submit, 오류, 복구"
  preview={<Stories.FieldValidationPreview />}
  sources={[{ label: 'React', language: 'tsx', code: String.raw`function EmailField() {
  const [submitted, setSubmitted] = useState(false);
  const [value, setValue] = useState('');
  const invalid = submitted && !value.includes('@');

  return (
    <Form noValidate onSubmit={(event) => { event.preventDefault(); event.currentTarget.checkValidity(); setSubmitted(true); }}>
      <Field.Root invalid={invalid}>
        <Field.Label>Alert email</Field.Label>
        <Field.Control
          aria-invalid={invalid || undefined}
          name="email"
          onChange={(event) => setValue(event.currentTarget.value)}
          required
          type="email"
          value={value}
        />
        <Field.Description>Used only for operational alerts.</Field.Description>
        {invalid ? <Field.Error match>{value ? 'Enter a valid email.' : 'Email is required.'}</Field.Error> : null}
      </Field.Root>
      <Button type="submit">Save email</Button>
    </Form>
  );
}` }]}
/>
