import { TRButton } from '@tinyrack/ui/components/button';
import { TRField } from '@tinyrack/ui/components/field';
import { TRCode } from '@tinyrack/ui/components/code';
import { TRTable } from '@tinyrack/ui/components/table';
import { tinyrackControlMetrics } from '@tinyrack/ui/core';

## 원칙

밀도 높은 보조 액션에는 `sm`, 기본 운영 인터페이스에는 `md`, 터치 편의나 강조가 필요할 때만 `lg`를 사용하세요.

## 선택 기준

- 기본 액션을 의도적으로 더 강조하는 경우가 아니라면 한 액션 행의 컨트롤은 같은 크기 레시피를 사용합니다.
- 포인터 사용이 예상되고 밀도가 중요하면 `sm`, 기본값은 `md`, 터치 중심이거나 강조가 큰 작업에는 `lg`를 사용합니다.
- 높이만 재정의하지 말고 높이, 안쪽 여백, 간격, 글자 크기, 행간을 포함한 전체 레시피를 함께 바꿉니다.

## 시각적 스케일

<div className="grid gap-tinyrack-md" role="list">
  {['sm', 'md', 'lg'].map((size) => (
    <section className="grid items-center gap-tinyrack-md border border-tinyrack-border p-tinyrack-lg md:grid-cols-[7rem_minmax(0,1fr)]" key={size} role="listitem">
      <div><strong>{size}</strong><div className="text-tinyrack-sm text-tinyrack-text-muted"><TRCode>{tinyrackControlMetrics[size].height}</TRCode></div></div>
      <div className="flex flex-wrap items-center gap-tinyrack-sm"><TRButton uiSize={size}>배포</TRButton><TRField.Root uiSize={size}><TRField.Control aria-label={`${size} 랙 이름`} placeholder="rack-a" /></TRField.Root></div>
    </section>
  ))}
</div>

</section>

## 상태와 터치 영역

스타일을 임의로 복제하지 말고 컴포넌트의 의미 기반 상태를 사용해요. 호버는 사용 가능성을 미리 보여 주고, 누름은 현재 입력을 나타내며, 선택은 결정을 유지해요. 비활성화는 사용할 수 없는 동작을 막고, 포커스는 키보드 탐색 중에도 보여야 해요. 호버만 단서로 사용하지 마세요. 터치 중심 레이아웃에서는 `lg`를 선택하거나 상호작용 영역을 줄이지 않은 채 컨트롤 주변 간격을 늘리세요.

## 구현

제품 전체에 다른 밀도가 필요하면 전역 치수를 재정의하고, 하나의 컴포넌트에 의도적인 예외가 필요하면 `--tr-*` 레시피 토큰을 사용하세요.

```css
:root {
  --tinyrack-control-height-md: 2.5rem;
}

.rack-toolbar {
  --tr-btn-height: 2.25rem;
}
```

## 접근성

모든 컨트롤에는 보이는 레이블 또는 접근 가능한 이름, 명확한 포커스 상태, 색상에만 의존하지 않는 상태 표현이 필요합니다. 컴포넌트의 키보드 동작을 유지하고 터치 중심 흐름에 더 큰 목표 영역이 필요하면 `lg`를 사용합니다. 보이는 표면만 키우고 실제 상호작용 영역을 그대로 두지 마세요.
