## 계약

| 항목 | 계약 |
| --- | --- |
| 제출 | 네이티브 `submit` 이벤트와 `FormData`를 사용하거나 `onFormSubmit`으로 파싱된 필드 값을 받습니다. |
| 유효성 검사 | 네이티브 validity와 Base UI `TRField.Error` 조합이 렌더링하는 외부 `errors`를 함께 사용합니다. |
| 초기화 | 네이티브 `type="reset"`은 비제어 값을 복원합니다. 제어 값과 애플리케이션 결과는 `onReset`에서 초기화해야 합니다. |
| 상태 | control은 비제어 또는 애플리케이션 제어 방식으로 사용할 수 있으며, 애플리케이션이 server error와 제출 결과를 소유합니다. |

import { TRForm } from '@tinyrack/ui/components/form';
import { TRInput } from '@tinyrack/ui/components/input';

function NativeRackForm() {
  const [submitted, setSubmitted] = useState('');

  return <TRForm
    aria-label="Native rack form"
    onReset={() => setSubmitted('')}
    onSubmit={(event) => {
      event.preventDefault();
      const value = String(new FormData(event.currentTarget).get('rack') ?? '');
      setSubmitted(value);
    }}
  >
    <label htmlFor="rack-name">
      Rack name
      <TRInput id="rack-name" name="rack" defaultValue="rack-alpha" />
    </label>
    <TRButton type="submit">Submit rack</TRButton>
    <TRButton type="reset">Reset form</TRButton>
    <output aria-live="polite">{submitted ? 'Submitted ' + submitted + '.' : ''}</output>
  </TRForm>;
}` }]}

import { TRButton } from '@tinyrack/ui/components/button';
import { TRForm } from '@tinyrack/ui/components/form';
import { TRInput } from '@tinyrack/ui/components/input';

function RequiredRackForm() {
  const [attempted, setAttempted] = useState(false);
  const [saved, setSaved] = useState('');
  const [value, setValue] = useState('');
  const invalid = attempted && value.trim().length === 0;

  return (
    <TRForm
      noValidate
      onSubmit={(event) => {
        event.preventDefault();
        setAttempted(true);
        if (event.currentTarget.checkValidity()) setSaved(value);
      }}
    >
      <label htmlFor="required-rack">Rack name</label>
      <TRInput
        aria-describedby={invalid ? 'required-rack-error' : undefined}
        aria-invalid={invalid || undefined}
        id="required-rack"
        name="rack"
        onChange={(event) => {
          setValue(event.currentTarget.value);
          setSaved('');
        }}
        required
        value={value}
      />
      {invalid ? <p id="required-rack-error" role="alert">Enter a rack name before saving.</p> : null}
      <TRButton type="submit">Save rack</TRButton>
      <output aria-live="polite">{saved ? 'Saved ' + saved + '.' : ''}</output>
    </TRForm>
  );
}` }]}
/>

import { TRButton } from '@tinyrack/ui/components/button';
import { TRField } from '@tinyrack/ui/components/field';
import { TRForm } from '@tinyrack/ui/components/form';

function ServerValidatedRackForm() {
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [result, setResult] = useState('');

  return <TRForm
    aria-label="Create rack form"
    errors={errors}
    onFormSubmit={(values) => {
      const rack = String(values.rack ?? '');
      if (rack.toLowerCase() === 'rack-alpha') {
        setErrors({ rack: 'Rack Alpha already exists.' });
        setResult('');
        return;
      }
      setErrors({});
      setResult('Created ' + rack + '.');
    }}
    onReset={() => { setErrors({}); setResult(''); }}
  >
    <TRField.Root name="rack">
      <TRField.Label>Rack name</TRField.Label>
      <TRField.Control defaultValue="rack-alpha" required />
      <TRField.Description>Use a name that is not already registered.</TRField.Description>
      <TRField.Error />
    </TRField.Root>
    <TRButton type="submit">Create rack</TRButton>
    <TRButton type="reset">Reset form</TRButton>
    <output aria-live="polite">{result}</output>
  </TRForm>;
}` }]}
/>

import { TRButton } from '@tinyrack/ui/components/button';
import { TRField } from '@tinyrack/ui/components/field';
import { TRForm, type TRFormActions } from '@tinyrack/ui/components/form';

export function RackForm() {
  const actionsRef = useRef<TRFormActions>(null);
  const [result, setResult] = useState('');

  return <TRForm<{ rack: string; region: string }>
    actionsRef={actionsRef}
    onFormSubmit={(values) => setResult('Validated ' + values.rack + ' in ' + values.region + '.')}
    validationMode="onBlur"
  >
    <TRField.Root name="rack">
      <TRField.Label>Rack name</TRField.Label>
      <TRField.Control required />
      <TRField.Error match="valueMissing">Enter a rack name.</TRField.Error>
    </TRField.Root>
    <TRField.Root name="region">
      <TRField.Label>Region</TRField.Label>
      <TRField.Control required />
      <TRField.Error match="valueMissing">Enter a region.</TRField.Error>
    </TRField.Root>
    <TRButton onClick={() => actionsRef.current?.validate('rack')} type="button">Validate rack</TRButton>
    <TRButton onClick={() => actionsRef.current?.validate()} type="button">Validate all</TRButton>
    <TRButton type="submit">Submit</TRButton>
    <output aria-live="polite">{result}</output>
  </TRForm>;
}` }]}

## API

`TRForm`은 native form props, ref, event, `className`, `style`, `render`를 전달하며 control은 자체 value를 계속 소유합니다. `TRFormProps<Values>`는 `onFormSubmit(values, details)`을 통해 field-value generic을 보존하고, event detail에는 `TRFormSubmitEventDetails`와 `TRFormSubmitEventReason`을 사용할 수 있습니다. server response에는 field name과 함께 `errors`를 사용하고 TRForm 또는 TRField 범위에서 `validationMode`를 선택하세요. submit을 만들지 않고 명시적으로 검증하려면 `TRFormActions.validate(name?)`를 사용합니다. native reset은 비제어 control만 복원하므로 제어 값은 애플리케이션의 `onReset` handler에서 초기화해야 합니다.
