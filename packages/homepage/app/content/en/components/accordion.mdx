## Contract

| Axis | Values | Default |
| --- | --- | --- |
| selection | controlled `value` or uncontrolled `defaultValue` | `[]` |
| expansion | single or `multiple` | single |
| availability | root `disabled` or item `disabled` | interactive |
| lifecycle | unmount, `keepMounted`, or `hiddenUntilFound` | unmount |

## Usage

<ComponentExampleTabs
  id="accordion-basic"
  title="Controlled multiple expansion"
  description="Open and close either section; the result below reflects the current value."
  previewLayout="stretch"
  preview={<Stories.AccordionInteractiveExample />}
  sources={[{ label: 'React', language: 'tsx', code: String.raw`import { useState } from 'react';
import { TRAccordion } from '@tinyrack/ui/components/accordion';

export function AccordionExample() {
  const [value, setValue] = useState<string[]>(['overview']);

  return (
    <div className="grid gap-3">
      <TRAccordion.Root multiple onValueChange={setValue} value={value}>
        <TRAccordion.Item value="overview">
          <TRAccordion.Header>
            <TRAccordion.Trigger>What is Tinyrack?</TRAccordion.Trigger>
          </TRAccordion.Header>
          <TRAccordion.Panel>A React-only UI system.</TRAccordion.Panel>
        </TRAccordion.Item>
        <TRAccordion.Item value="install">
          <TRAccordion.Header>
            <TRAccordion.Trigger>How do I install it?</TRAccordion.Trigger>
          </TRAccordion.Header>
          <TRAccordion.Panel>Install the package and component CSS.</TRAccordion.Panel>
        </TRAccordion.Item>
      </TRAccordion.Root>
      <output aria-live="polite">
        Expanded: {value.length === 0 ? 'none' : value.join(', ')}
      </output>
    </div>
  );
}` }]}
/>

## API

Base UI owns expansion, controlled and uncontrolled state, trigger semantics, and panel relationships. TRAccordion triggers remain in normal document tab order; the deprecated arrow-key roving-focus props are intentionally not part of this guidance.
